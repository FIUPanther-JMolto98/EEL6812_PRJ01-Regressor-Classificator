model_checkpoint_callback = ModelCheckpoint(
    filepath="regmodl03.keras",
    save_best_only=True,
    monitor='val_loss',
    mode='min'
)
early_stopping_callback = EarlyStopping(
    monitor='val_loss',
    mode='min',
    patience=500,
    verbose=1,
    restore_best_weights=True
)
